SELECT  yinxiang.dbo.t_order.createtime ,
        storeid ,
        store ,
        yinxiang.dbo.t_product.code ,
        yinxiang.dbo.t_product.name ,
        yinxiang.dbo.t_product.spec ,
        yinxiang.dbo.t_product.unit ,
        yinxiang.dbo.t_orderdetail.num,
        yinxiang.dbo.t_orderdetail.price ,
        yinxiang.dbo.t_orderdetail.amount
        
FROM    yinxiang.dbo.t_order
        inner JOIN yinxiang.dbo.t_store ON yinxiang.dbo.t_order.storeid = yinxiang.dbo.t_store.id
        INNER JOIN yinxiang.dbo.t_orderdetail ON yinxiang.dbo.t_order.id = yinxiang.dbo.t_orderdetail.mainid
        INNER JOIN yinxiang.dbo.t_product ON yinxiang.dbo.t_orderdetail.productno = yinxiang.dbo.t_product.id
WHERE   yinxiang.dbo.t_order.createtime >= '********'
        AND yinxiang.dbo.t_order.createtime <= '########'
