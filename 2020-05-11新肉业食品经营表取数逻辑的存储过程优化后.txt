                    部门名称 ,
                    ROUND(SUM(数量), 2) 数量 ,
                    ROUND(SUM(毛利), 2) 毛利
            INTO    #gk
            FROM    #rsjyb2
            GROUP BY 部门代码 ,
                    部门名称;
 
            UPDATE  t
            SET     t.本月累计销量 = k.数量 ,
                    t.本月累计毛利 = k.毛利
            FROM    #ttt t
                    INNER JOIN #gk k ON t.部门代码 = k.部门代码
                                        AND t.部门名称 = k.部门名称;
 
            DECLARE @本月累计销量 FLOAT ,
                @本月累计毛利 FLOAT ,
                @当天屠宰头数 FLOAT ,
                @当月屠宰头数 FLOAT;
 
            SELECT  @本月累计销量 = SUM(本月累计销量) ,
                    @本月累计毛利 = SUM(本月累计毛利)
            FROM    #ttt;
 
            SELECT  @当天屠宰头数 = FDayHeadNum
            FROM    yx_rs_DayHeadNum
            WHERE   FDATE = @EndDate1;
 
            SELECT  @当月屠宰头数 = SUM(FDayHeadNum)
            FROM    yx_rs_DayHeadNum
            WHERE   FDATE BETWEEN @BegDate1 AND @EndDate1;
 
            UPDATE  #ttt
            SET     本月累计销量 = ROUND(@本月累计销量, 2) ,
                    本月累计毛利 = ROUND(@本月累计毛利, 2) ,
                    当天屠宰头数 = @当天屠宰头数 ,
                    当月屠宰头数 = @当月屠宰头数
            WHERE   日期 = '合计';
 

        END; 



    SELECT  *
    FROM    #ttt;
